// Git configuration local to a project
// path - .git/config
git config --local --list|-l	list all configuration 

// Git configuration for a specific user
// path - ~/.gitconfig
git config --global --list|-l 	list all configuration

// Git configuration for all users on a system
// path - <path>/etc/gitconfig
git config --system --list|-l 	list all configuration

git config --list|-l [--show-origin]    to view configurations from all three places, optionally we can
                                        display source of each configuration with --show-origin

git config [--global|--system] --edit|-e	opens config file in an editor
git config [--global|--system] <property-name>	show value of property-name
