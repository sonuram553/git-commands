// Git configuration local to a project
// path - .git/config
git config --local --list|-l	list all configuration 

// Git configuration for a specific user
// path - ~/.gitconfig
git config --global --list|-l 	list all configuration

// Git configuration for all users on a system
// path - <path>/etc/gitconfig
git config --system --list|-l 	list all configuration

// To view configurations from all three places, optionally we can
// display source of each configuration with --show-origin
git config --list|-l [--show-origin]    

git config [--local|--global|--system] --edit|-e	opens config file in an editor
git config [--local|--global|--system] <property-name> [<value>]	show or set value of the property-name

// A property can appear in multiple config files. Display which file Git is using
// to set the value of the property 
git config --show-origin <property-name>
